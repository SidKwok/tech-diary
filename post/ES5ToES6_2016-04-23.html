ES5转ES6总结

2016.04.23

最近好像有点把人品败光了。唉，不知道网易和daocloud那边什么情况，转运桃子我已经发过了，希望它能够给我带来一点运气。但是也不能荒废嘛，下面对ES5的一些用法转成ES6的，至于相关解释就不说了，因为会记住的，记不住就看一下书。不定期更新。

上为ES5，下为ES6。

将数组转为函数参数

// ES5

function f(x, y, z){}
var args = [0, 1, 2];
f.apply(null, args);

// ES6

function f(x, y, z){}
var args = [0, 1, 2];
f(...args);

参数默认值

// ES5

function f(x, y){
  y = y || "world";
}

// 这样子有个缺点是没有办法检测false和''等一系列转化之后为false的值

// ES6

function f(x, y = "world"){};

合并数组

// ES5

var arr1 = ['1', '2'];
var arr2 = ['3', '4'];
var arr3 = ['5', '6'];

var arr4 = arr1.concat(arr2, arr3);

// ES6
...

var arr4 = [...arr1, ...arr2, ...arr3];

将伪数组转化为真数组

// ES5

function f(){
  var arg = Array.prototype.apply(arguments);
}

// ES6

function f(){
  var arg = [...arguments];
}

箭头函数

// ES5

var f1 = function(a){
  return a;
}

var f2 = function(){
  return 5;
}

var f3 = function(){
  return {a:1, b:2};
}

var f4 = function(a){
  a++;
  return a;
}

// ES6

var f1 = a => a;
var f2 = () => 5;
var f3 = () => ({a:1, b:2});
var f4 = a => {a++;return a;};
